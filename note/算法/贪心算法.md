1. 买卖股票的最佳时机

- 找到最小的一天，记录下来，找到之后相差最大的一天
`
func maxProfit(prices []int) (res int) {
    minV := prices[0]
    for _, v := range prices {
        res = max(res, v-minV)
        minV = min(minV, v)
    }
    return
}
